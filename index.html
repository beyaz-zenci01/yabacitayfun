function updateNearbyMahalleler() {

const mainMahalle = document.getElementById("main-mahalle").value;

const nearbyMahalleler = mahalleYakınlıkBilgisi[mainMahalle] || [];

const nearbyList = document.createElement("ul");

nearbyList.classList.add("info");

nearbyMahalleler.forEach(mahalle => {

const listItem = document.createElement("li");

listItem.classList.add("underline");

listItem.textContent = mahalle;

nearbyList.appendChild(listItem);

});

const nearbyContainer = document.getElementById("nearby-mahalleler");

nearbyContainer.innerHTML = "";

if (nearbyMahalleler.length > 0) {

const infoText = document.createElement("p");

infoText.textContent = "İlk paket bu mahalleler ile aynı yöndedir";

infoText.classList.add("info");

nearbyContainer.appendChild(infoText);

nearbyContainer.appendChild(nearbyList);

}

}

function checkYakınlık() {

const mainMahalle = document.getElementById("main-mahalle").value;

const secondaryMahalle = document.getElementById("secondary-mahalle").value;

const resultContainer = document.getElementById("yakınlık-sonucu");

resultContainer.innerHTML = "";

if (!mainMahalle || !secondaryMahalle) {

const errorText = document.createElement("p");

errorText.textContent = "Lütfen her iki mahalleyi de seçin.";

errorText.classList.add("error");

resultContainer.appendChild(errorText);

return;

}

if (mainMahalle === secondaryMahalle) {

const errorText = document.createElement("p");

errorText.textContent = "Aynı mahalleyi seçtiniz, lütfen farklı mahalleler seçin.";

errorText.classList.add("error");

resultContainer.appendChild(errorText);

return;

}

const nearbyMahalleler = mahalleYakınlıkBilgisi[mainMahalle] || [];

const isYakın = nearbyMahalleler.includes(secondaryMahalle);

const resultText = document.createElement("p");

if (isYakın) {

resultText.textContent = ${mainMahalle} ve ${secondaryMahalle} birbirine yakındır.;

resultText.classList.add("success");

} else {

resultText.textContent = ${mainMahalle} ve ${secondaryMahalle} birbirine yakın değildir.;

resultText.classList.add("warning");

}

resultContainer.appendChild(resultText);

kullanımSayacı++;

localStorage.setItem("kullanımSayacı", kullanımSayacı);

document.getElementById("kullanım-sayacı").textContent = Bu sistem ${kullanımSayacı} kere kullanıldı.;

}
